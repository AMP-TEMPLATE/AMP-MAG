# Step 2: Adding Scripts and Styles with <head></head> tags.
<script async src="https://cdn.ampproject.org/v0.js"></script> <!-- Basic amp script -->
<link rel="stylesheet" type="text/css" href="https://fonts.googleapis.com/css?family=Roboto"> <!-- Set basic font for the page. -->
<link rel="stylesheet" type="text/css" href="custom.css"> <!-- Link your custom stylesheet -->
 <meta name="viewport" content="width=device-width,minimum-scale=1,initial-scale=1"> <!-- This ensures that the page is responsive in different types of screens. -->
